int main()
{
	int n, m;
	cin >> n >> m;
	if (m > n)
		m = m - n;
	vector<int> v(n+m);
	for (int i=m;i<n+m;i++)
	{
		int x;
		cin >> x;
		v[i] = x;
	}
	for (int i = n; i < n + m; i++)
	{
		int temp = v[i];
		v[i - n] = temp;
	}
	for (int i = 0; i < n - 1; i++)
		cout << v[i] << " ";
	cout << v[n-1];

	return 0;
}

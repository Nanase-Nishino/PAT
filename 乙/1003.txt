#include<stdio.h>
#include<iostream>
#include<string>
#include<sstream>
#include<vector>
#include<array>
using namespace std;

//空字符不是空格啊 注意PA
bool judge(string s)
{
	vector<int> a(3, 0);
	int n = s.size();
	if (n == 0)
		return false;
	int i = 0;
	for (;(s[i] != 'P');)
	{
		if (s[i] == 'A')
		{
			a[0]++;
			i++;
		}
		else
			return false;
	}
	for (i=i+1;(s[i] != 'T');)
	{
		if (s[i] == 'A')
		{
			a[1]++;
			i++;
		}
		else
			return false;
	}
	for (i = i + 1; i < n ;)
	{
		if (s[i] == 'A')
		{
			a[2]++;
			i++;
		}
		else
			return false;
	}
	if (a[1] == 0)
		return false;
	if (a[0] * a[1] == a[2])
		return true;
	return false;
}
int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		string s;
		cin >> s;
		
		if (judge(s))
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	system("pause");
	return 0;
}
